import java.util.Scanner; public class Sum_Of_Digit {
public static void main(String[] args) {

int n
int sum=0;
int rem;
System.out.println("Enter a number");
n=sc.nextInt();
for(i=n;i>0;i=i/10){
rem=i%10;
sum=sum+rem;
}
System.out.println("Sum of digit of number" +sum);
}
